## Real-World Example

```rust
// Professional automation script example
info("🚀 Initializing SnapRun automation workflow...");

// Collect user requirements
let project_name = ask_input(
    "Project Setup", 
    "Enter your project name:",
    "my-awesome-project"
);

let project_type = ask_select(
    "Project Type",
    "Select your project type:",
    ["Web Application", "Desktop App", "Mobile App", "API Service"]
);

// Create project structure
let project_data = #{
    name: project_name,
    type: project_type,
    created: timestamp(),
    version: "1.0.0",
    author: get_user_name()
};

// Setup project directory
let project_path = get_home_dir() + "/Projects/" + project_name;
create_dir(project_path);

// Generate configuration
write_file(
    project_path + "/config.json", 
    to_json(project_data, true)  // Pretty print JSON
);

// Create README with dynamic content
let readme_content = `# ${project_name}

## Project Information
- **Type:** ${project_type}
- **Created:** ${project_data.created}
- **Version:** ${project_data.version}

## Getting Started
Follow the setup instructions to get started with your ${project_type.to_lower()}.

---
*Generated by SnapRun Automation*
`;

write_file(project_path + "/README.md", readme_content);

// Success notification
render_html("Project Created Successfully", `
<div style="
    padding: 2rem;
    background: linear-gradient(135deg, #10B981, #059669);
    color: white;
    border-radius: 12px;
    font-family: -apple-system, BlinkMacSystemFont, 'Segoe UI', Roboto, sans-serif;
">
    <h2 style="margin: 0 0 1rem 0; font-size: 1.5rem;">✅ Project Setup Complete!</h2>
    <p style="margin: 0; opacity: 0.9;">
        <strong>${project_name}</strong> has been created successfully at:<br>
        <code style="background: rgba(255,255,255,0.2); padding: 0.25rem 0.5rem; border-radius: 4px;">
            ${project_path}
        </code>
    </p>
</div>
`);

info("✨ Automation workflow completed successfully!");
```
